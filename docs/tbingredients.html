    • Tabla ingredients

    A continuacion la table que almacenara la informacion de los usuarios solo para login.

    Nombre del Campo Tipo del Campo
    id_ingredient INT NOT NULL auto_increment PRIMARY KEY
    ingredient_name VARCHAR(30) NOT NULL
    ingredient_description VARCHAR(300) NOT NULL
    id_ingredient_type INT NOT NULL
    id_ingredient_galery INT NOT NULL
    id_ingredient_followers INT NOT NULL
    id_ingredient_vote INT NOT NULL
    id_ingredient_comments INT NOT NULL
    id_ingredient_qa INT NOT NULL
    FOREIGN KEY (id_user) REFERENCES users(id_user)
    FOREIGN KEY (id_ingredients_type) REFERENCES ingredients_types(id_ingredient_type)
    FOREIGN KEY (id_ingredients_galery) REFERENCES ingredients_galeries(id_ingredient_galery)
    FOREIGN KEY (id_ingredients_followers) REFERENCES ingredients_followers(id_ingredient_followers)
    FOREIGN KEY (id_ingredients_vote) REFERENCES ingredients_votes(id_ingredient_vote)
    FOREIGN KEY (id_ingredients_comments) REFERENCES ingredients_comments(id_ingredient_comments)
    FOREIGN KEY (id_ingredients_qa) REFERENCES ingredients_qa(id_ingredient_qa)
